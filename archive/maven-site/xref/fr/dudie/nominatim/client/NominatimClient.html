14  package fr.dudie.nominatim.client;
15  
16  import java.io.IOException;
17  import java.util.List;
18  
19  import fr.dudie.nominatim.model.Address;
20  
21  /**
22   * Interface to use the Google Maps Service.
23   * 
24   * @author Jérémie Huchet
25   */
26  public interface NominatimClient {
27  
28      /**
29       * Search for results with the given query.
30       * 
31       * @param query
32       *            the query
33       * @return a list of results
34       * @throws IOException
35       *             a communication error occurred
36       */
37      List<Address> search(String query) throws IOException;
38  
39      /**
40       * Reverse geocode the given coordinates.
41       * 
42       * @param longitude
43       *            a longitude
44       * @param latitude
45       *            a latitude
46       * @return an address corresponding to the given longitude and latitude or <code>null</code> if
47       *         no result found
48       * @throws IOException
49       *             a communication error occurred
50       */
51      Address getAddress(double longitude, double latitude) throws IOException;
52  
53      /**
54       * Reverse geocode the given coordinates using a specific zoom level
55       * 
56       * @param longitude
57       *            a longitude
58       * @param latitude
59       *            a latitude
60       * @param zoom
61       *            a osm zoom level
62       * @return an address corresponding to the given longitude and latitude or <code>null</code> if
63       *         no result found
64       * @throws IOException
65       *             a communication error occurred
66       */
67      Address getAddress(double longitude, double latitude, int zoom) throws IOException;
68  
69      /**
70       * A convenience method to do the same as {@link #getAddress(double, double)} but with int E6
71       * latitude and longitude.
72       * 
73       * @param longitudeE6
74       *            a longitude E6
75       * @param latitudeE6
76       *            a latitude E6
77       * @return an address corresponding to the given longitude and latitude or <code>null</code> if
78       *         no result found
79       * @throws IOException
80       *             a communication error occurred
81       */
82      Address getAddress(int longitudeE6, int latitudeE6) throws IOException;
83  
84      /**
85       * Reverse geocode the given OSM id.
86       * 
87       * @param type
88       *            An OSM type [N|W|R]
89       * @param id
90       *            An OSM id
91       * @return an address corresponding to the given osm type and id pair or <code>null</code> if no
92       *         result found
93       * @throws IOException
94       */
95      Address getAddress(String type, long id) throws IOException;
96  
97  }
