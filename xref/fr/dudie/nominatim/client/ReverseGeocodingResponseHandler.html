14  package fr.dudie.nominatim.client;
15  
16  import java.io.IOException;
17  
18  import org.apache.http.HttpResponse;
19  import org.apache.http.client.ResponseHandler;
20  import org.apache.http.util.EntityUtils;
21  import org.json.JSONException;
22  import org.json.JSONObject;
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;
25  
26  import fr.dudie.nominatim.model.Address;
27  
28  /**
29   * Parses a json response from the Nominatim API for a reverse geocoding request.
30   * 
31   * @author Jérémie Huchet
32   */
33  public final class ReverseGeocodingResponseHandler implements ResponseHandler<Address> {
34  
35      /** The event logger. */
36      private static final Logger LOGGER = LoggerFactory
37              .getLogger(ReverseGeocodingResponseHandler.class);
38  
39      /**
40       * {@inheritDoc}
41       * 
42       * @see org.apache.http.client.ResponseHandler#handleResponse(org.apache.http.HttpResponse)
43       */
44      @Override
45      public Address handleResponse(final HttpResponse response) throws IOException {
46  
47          String content = null;
48          Address address = null;
49          try {
50              content = EntityUtils.toString(response.getEntity(), "utf-8");
51              final JSONObject json = new JSONObject(content);
52              address = NominatimJsonUtils.toAddress(json);
53          } catch (final JSONException e) {
54              throw new IOException("Unable to parse the json response received from Nominatim:\n"
55                      + content);
56          }
57  
58          return address;
59      }
60  }
